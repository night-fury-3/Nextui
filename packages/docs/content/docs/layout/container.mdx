# Container

A component for fixing an element's width to the current breakpoint.

<Playground
  title="Default"
  desc="Container which sets a `max-width` and a default `gap` at each responsive breakpoint"
  code={`<Container>
<ExampleBlock>
<Row justify="center" align="center">
  <Text h6 size={15} color="white" style={{ margin: 0 }}>
    NextUI gives you the best developer experience with all the features you need for building beautiful and modern websites and applications.
  </Text>
</Row>
  </ExampleBlock>
</Container>
`}
/>

<Playground
  title="Fluid"
  desc="Container which its `width: 100%` at all breakpoints"
  code={`<Container fluid>
<ExampleBlock>
<Row justify="center" align="center">
  <Text h6 size={15} color="white" style={{ margin: 0 }}>
    NextUI gives you the best developer experience with all the features you need for building beautiful and modern websites and applications.
  </Text>
</Row>
  </ExampleBlock>
</Container>
`}
/>

<Playground
  title="Breakpoints"
  desc="Container which is `width: 100%` until the specified breakpoint"
  code={`
  () => {
   const Block = () => (<ExampleBlock>
  <Row justify="center" align="center">
    <Text h6 size={15} color="white" style={{ margin: 0 }}>
      NextUI gives you the best developer experience with all the features you
      need for building beautiful and modern websites and applications.
    </Text>
  </Row>
</ExampleBlock>
);
    return (
    <>
    <Container xl>
        <Block />
    </Container>
    <Spacer y={1}/>
    <Container lg>
        <Block />
    </Container>
    <Spacer y={1}/>
    <Container md>
        <Block />
    </Container>
    <Spacer y={1}/>
    <Container sm>
        <Block />
    </Container>
    <Spacer y={1}/>
    <Container xs>
        <Block />
    </Container>
    </>
    );
 }
 `}
/>
