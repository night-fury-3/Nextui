# Container

A component for fixing an element's width to the current breakpoint.

```jsx
import { Container, Row, Col } from '@nextui-org/react';
```

### Custom breakpoints

Override the default `@nextui-org/react` breakpoints.

```tsx
const breakpoints: NextUIThemesBreakpoints = {
  // Extra small devices (phones)
  xs: { min: '0', max: '650px' },
  // Small devices (tablets)
  sm: { min: '650px', max: '960px' },
  // Medium devices (landscape tablets)
  md: { min: '960px', max: '1280px' },
  // Large devices (desktops)
  lg: { min: '1280px', max: '1400px' },
  // Extra large devices (larger desktops)
  xl: { min: '1400px', max: '1920px' },
};

const App = () => (
  <ThemeProvider theme={{ breakpoints: breakpoints }}>
    <CssBaseline />
    <AppComponent />
  </ThemeProvider>
);
```

#### Default breakpoints

| Breakpoint  | <650px | ≥650px | ≥960px | ≥1280px | ≥1400px |
| ----------- | ------ | ------ | ------ | ------- | ------- |
| **default** | `100%` | 650px  | 960px  | 1280px  | 1400px  |
| **xs**      | `100%` | 650px  | 960px  | 1280px  | 1400px  |
| **sm**      | `100%` | `100%` | 960px  | 1280px  | 1400px  |
| **md**      | `100%` | `100%` | 960px  | 1280px  | 1400px  |
| **lg**      | `100%` | `100%` | `100%` | 1280px  | 1400px  |
| **xl**      | `100%` | `100%` | `100%` | `100%`  | 1400px  |
| **fluid**   | `100%` | `100%` | `100%` | `100%`  | `100%`  |

### APIs

<Spacer y={2} />

#### Container Props

| Attribute        | Type             | Accepted values                | Description                      | Default |
| ---------------- | ---------------- | ------------------------------ | -------------------------------- | ------- |
| **fluid**        | `boolean`        | `true / false`                 | `width: 100%` at all breakpoints | `false` |
| **xs**           | `boolean`        | `true / false`                 | Extra small devices (<650px)     | `false` |
| **sm**           | `boolean`        | `true / false`                 | Small devices (≥650px)           | `false` |
| **md**           | `boolean`        | `true / false`                 | Medium devices (≥960px)          | `false` |
| **lg**           | `boolean`        | `true / false`                 | Large devices (≥1280px)          | `false` |
| **xl**           | `boolean`        | `true / false`                 | Extra large devices (≥1400px)    | `false` |
| **gap**          | `number`         | -                              | Children's spacing               | 2       |
| **wrap**         | `Wrap`           | [Wrap](#wrap)                  | CSS "flex-wrap"                  | `wrap`  |
| **display**      | `Display`        | [Display](#display)            | CSS "display"                    | `block` |
| **justify**      | `Justify`        | [Justify](#justify)            | CSS "justify-content"            | -       |
| **direction**    | `Direction`      | [Direction](#direction)        | CSS "flex-direction"             | -       |
| **alignItems**   | `AlignItems`     | [AlignItems](#align-items)     | CSS "align-items"                | -       |
| **alignContent** | `AlignContent`   | [AlignContent](#align-content) | CSS "align-content"              | -       |
| **as**           | `ElementType`    | [ElementType](#element-type)   | Root node component              | `div`   |
| ...              | `HTMLAttributes` | `'id', 'className', ...`       | Native element props             | -       |

### Container types

<Spacer y={1} />

#### Wrap

```ts
type Wrap = 'nowrap' | 'wrap' | 'wrap-reverse';
```

#### Display

```ts
type Display =
  | 'flex'
  | 'block'
  | 'grid'
  | 'inline'
  | 'inline-block'
  | 'inline-flex'
  | 'inline-grid';
```

#### Justify

```ts
type Justify =
  | 'flex-start'
  | 'center'
  | 'flex-end'
  | 'space-between'
  | 'space-around'
  | 'space-evenly';
```

#### Direction

```ts
type Direction = 'row' | 'row-reverse' | 'column' | 'column-reverse';
```

#### Align Items

```ts
type AlignItems = 'flex-start' | 'flex-end' | 'center' | 'stretch' | 'baseline';
```

#### Align Content

```ts
type AlignContent =
  | 'stretch'
  | 'center'
  | 'flex-start'
  | 'flex-end'
  | 'space-between'
  | 'space-around';
```

#### Element Type

```ts
keyof JSX.IntrinsicElements;
```
