# Pagination

The Pagination component allows you to display active page and navigate between multiple pages.

```jsx
import { Pagination } from '@nextui-org/react';
```

  title="Colors"
  desc="You can set a starting page with the `initialPage` prop."

const [selectedColor, setSelectedColor] = React.useState('primary');
const colors = ['primary', 'secondary', 'success', 'warning', 'error', '#f4a']
const capitalize = (str) => {
  const lower = str.toLowerCase();
  return str.charAt(0).toUpperCase() + lower.slice(1);
}
return(
<Grid.Container gap={2}>
    <Grid xs={12}>
      <Pagination color={selectedColor} total={10} />
    </Grid>
    <Grid xs={12}>
        <Radio.Group row size="small" value={selectedColor} onChange={setSelectedColor}>
            {colors.map((color) => (
                <Radio key={color} value={color} color={color}>
                    {capitalize(color)}
                </Radio>
            ))}
        </Radio.Group>
    </Grid>    
</Grid.Container>
)}  
`}
/>
